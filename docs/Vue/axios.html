    vue3 axios的封装

    张文杰 7/4/2022 Vue3

    • 安装 axios

      • npm install axiso

    • 调用模拟数据

      • https://jsonplaceholder.typicode.com/

    • 封装

      • src/api/index.js
    import axios from "axios";
import request from "./request.js";

// 创建axios实例
const instance = axios.create({
  baseURL: "https://jsonplaceholder.typicode.com",
  timeout: 5000,
});

// 请求拦截器
instance.interceptors.request.use(
  (config) => {
    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);

// 响应拦截器
instance.interceptors.response.use(
  (response) => {
    return response.data;
  },
  (error) => {
    return Promise.reject(error);
  }
);

const Http = {}; //包裹请求方法的容器

for (let key in request) {
  let api = request[key]; //url和method
  Http[key] = async function (
    params, //请求参数如果是get则为url,put/post/patch则为data,delete则为url
    isFormData = false, //标识是否为form-data请求
    config = {} //配置参数
  ) {
    let url = api.url;
    let newParams = {}; //这里作判断content-type是否为form-data请求的判断
    if (params && isFormData) {
      newParams = new FormData(); //创建formdata格式的对象
      for (let i in params) {
        newParams.append(key, params[key]);
      }
    } else {
      newParams = params;
    }
    let response = {}; //声明请求的返回值
    //不同请求的判断
    if (api.method == "put" || api.method == "post" || api.method == "patch") {
      //类似于axios.get(url,params,config)
      //请求.请求方式(路径，参数，配置)
      //await instance[api.method](api.url,newParams,config)
      // loadingInstance = ElLoading.request({
      //     fullscreen: true
      // });
      try {
        response = await instance[api.method](api.url, newParams, config);
        // loadingInstance.close()
      } catch (err) {
        response = err;
        // loadingInstance.close()
      }
    } else if (api.method == "delete" || api.method == "get") {
      // loadingInstance = ElLoading.request({
      //     fullscreen: true
      // });
      //注意delete请求和get请求中的params是放在config中的，所以这里要重新赋值
      config.params = newParams;
      try {
        response = await instance[api.method](api.url, config);
        // loadingInstance.close()
      } catch (err) {
        response = err;
        // loadingInstance.close()
      }
    }
    return response; //返回请求的响应值
  };
}
export default Http;

    • 使用
    <script>
import { getCurrentInstance } from 'vue';
export default {
  setup(props, context) {
    const { proxy } = getCurrentInstance();
    proxy.$http.getComments().then((res) => {
      console.log(res);
    });

    setTimeout(() => {
      proxy.$http.getPosts().then((res) => {
        console.log(res);
      });
    }, 3000);
  },
};
</script>
